\iftoggle{todo_toggle}{
  \newlength{\todoshift}%
  \setlength{\todoshift}{2cm} % change this to control how far into
  % the margin it sticks
  \newenvironment{todo}
  {%
    \begin{adjustwidth}{-\todoshift}{0pt}%
      \begin{todo_box}\noindent\textbf{TODO:}\quad
      }
      {%
      \end{todo_box}
    \end{adjustwidth}\par
  }
}{
  \excludecomment{todo}
}

  {

  \end{emphasis}\par}
}{

}
